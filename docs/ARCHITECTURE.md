```python
from abc import ABC, abstractmethod

class BaseScraper(ABC):
    """Abstract base class for all scrapers"""
    
    def __init__(self, config: ScraperConfig):
        self.config = config
        self.rate_limiter = RateLimiter(config.rate_limit)
        self.proxy_manager = ProxyManager(config.proxy)
        self.browser: Browser = None
    
    async def __aenter__(self):
        await self.setup()
        return self
    
    async def __aexit__(self, *args):
        await self.teardown()
    
    @abstractmethod
    async def scrape(self, **kwargs) -> ScrapedData:
        """Main scraping method - implement in subclass"""
        pass
    
    @abstractmethod
    async def parse(self, html: str) -> ParsedData:
        """Parse HTML content - implement in subclass"""
        pass
    
    async def fetch(self, url: str) -> str:
        """Fetch URL with rate limiting and retries"""
        await self.rate_limiter.acquire()
        
        for attempt in range(self.config.max_retries):
            try:
                proxy = await self.proxy_manager.get_proxy()
                response = await self._make_request(url, proxy)
                return response
            except Exception as e:
                if attempt == self.config.max_retries - 1:
                    raise
                await asyncio.sleep(self.config.retry_delay)
```

### 5.2 Indexing Strategy

```sql
-- Performance-critical indexes

-- Matches: Frequently queried by date and status
CREATE INDEX CONCURRENTLY idx_matches_date_status 
ON matches(match_date, status) 
WHERE status IN ('scheduled', 'live');

-- Predictions: Quick lookup by match
CREATE INDEX CONCURRENTLY idx_predictions_match_model 
ON predictions(match_id, model_name);

-- Odds: Time-series queries
CREATE INDEX CONCURRENTLY idx_odds_match_time 
ON odds(match_id, recorded_at DESC);

-- Value bets: Active bets lookup
CREATE INDEX CONCURRENTLY idx_value_bets_active 
ON value_bets(match_id, result) 
WHERE result = 'pending';

-- Full-text search on news
CREATE INDEX CONCURRENTLY idx_news_search 
ON news USING gin(to_tsvector('english', title || ' ' || content));
```

### 5.3 Partitioning Strategy

```sql
-- Partition matches by season for better performance
CREATE TABLE matches (
    id SERIAL,
    match_date TIMESTAMPTZ NOT NULL,
    season VARCHAR(10) NOT NULL,
    -- other columns
) PARTITION BY LIST (season);

CREATE TABLE matches_2324 PARTITION OF matches FOR VALUES IN ('2023-24');
CREATE TABLE matches_2425 PARTITION OF matches FOR VALUES IN ('2024-25');

-- Partition odds by month (high-volume table)
CREATE TABLE odds (
    id SERIAL,
    recorded_at TIMESTAMPTZ NOT NULL,
    -- other columns
) PARTITION BY RANGE (recorded_at);

CREATE TABLE odds_2025_01 PARTITION OF odds 
FOR VALUES FROM ('2025-01-01') TO ('2025-02-01');
```
